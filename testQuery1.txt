# Prefixes
PREFIX dc:  &lthttp://purl.org/dc/elements/1.1/&gt
PREFIX rdf: &lthttp://www.w3.org/1999/02/22-rdf-syntax-ns#&gt
PREFIX rdfs: &lthttp://www.w3.org/2000/01/rdf-schema#&gt
PREFIX mso: &lthttp://eatld.et.tu-dresden.de/mso/&gt
PREFIX sce: &lthttp://eatld.et.tu-dresden.de/sce/&gt



# CenterView Attribute List
SELECT DISTINCT ?propertyLabel ?propertyValue ?propertyUnit
WHERE {
    {
        sce:a23411 ?property ?propertyValue.
        FILTER(isLiteral(?propertyValue))
    }
    UNION{
        sce:a23411 ?property ?quantity.
        ?quantity mso:numericalValue ?propertyValue;
                    mso:physicalUnit   ?propertyUnit.
    }
    ?property rdfs:label ?propertyLabel.
    FILTER (lang(?propertyLabel)='de')
}
