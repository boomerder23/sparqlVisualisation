
SELECT DISTINCT ?propertyLabel ?propertyValue ?propertyUnit
WHERE {
    {
        sce:a23411 ?property ?propertyValue.
        FILTER(isLiteral(?propertyValue))
    }
    UNION{
        sce:a23411 ?property ?quantity.
        ?quantity mso:numericalValue ?propertyValue;
                    mso:physicalUnit   ?propertyUnit.
    }
    ?property rdfs:label ?propertyLabel.
    FILTER (lang(?propertyLabel)='de')
}
